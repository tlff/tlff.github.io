    # codepush的用法

    首先安装code-push-cli@2.1.9 其他版本的和自己搭建的服务器配合不上

    1. 登录 code-push login http://192.168.18.44:3000 code-push logout
    2. 初始化一个应用 code-push app add renai Android react-native
    3. 获取应用的key code-push deployment ls renai -k │ Name │ Deployment Key │ ├────────────┼──────────────────────.─────────────────┤. │ Production │ IpS17r7OTZ6rhr39Jz5HYxNxcw0P4ksvOXqog │ ├────────────┼───────────────────────────────────────┤ │ Staging │ DgZdMfWE2QbWxuyS598y0uvy1Byj4ksvOXqog
    4. 推送一个更新 code-push release-react renai android -d Production --t 1.0.1 --des "" code-push release-react renai android -d Staging --t 1.0.1 --des ""
    5. 回滚一个更新 code-push rollback renai Production -r v2
    6. 禁用 code-push patch renai Production -l v1 --des "禁用" -x
    7. 查看历史 code-push deployment history BorrowApp Production

    • js调用codepush更新 附带一个文件

    直接调用update方法就行了

    import CodePush from 'react-native-code-push';
import {Toast} from '../components/layout';
import {dev} from '../config/config';
import {updateVersion, saveLog} from '../api/index';
// const dev=__DEV__
//为了测试方便 正式环境和测试环境的key改成一样的 区别在服务器部署位置,测试时使用代理将软件的链接全部代理到测试服务器
let key = dev
    ? 'aptbm5Ko2Ep7FAmvSsl6vN7EANJm4ksvOXqog'
    : 'aptbm5Ko2Ep7FAmvSsl6vN7EANJm4ksvOXqog';
export async function getInfo() {
    return CodePush.checkForUpdate(key);
}
export async function notifyAppReady() {
    return CodePush.notifyAppReady().then((e: any) => {
        if (e && e.package) {
            updateVersion(e.package?.appVersion, e.package?.label);
        }
    });
}
export async function update(dialog = false) {
    console.log('触发更新');
    saveLog('检查更新');
    let da = {
        mandatoryContinueButtonLabel: '强制更新/确认',
        optionalIgnoreButtonLabel: '取消',
        optionalInstallButtonLabel: '安装',
        title: '发现新版本',
        optionalUpdateMessage: '有新版本可用是否更新?',
    };
    const onStatus = (status) => {
        let maps = [
            '已经是最新版本',
            '安装完成等待生效',
            '忽略更新',
            '发生未知错误',
            '正在下载',
            '查询更新',
            '弹出确认提示框',
            '下载中',
            '下载完成',
        ];
        if (status === 0 || status === 5) {
            Toast(maps[status]);
        }
    };
    await CodePush.sync(
        // 第一个参数吗，是个对象，可定义更新的动作
        {
            // 安装模式 'IMMEDIATE' 立刻安装， ON_NEXT_RESUME 下次启动安装
            installMode: CodePush.InstallMode.IMMEDIATE,
            deploymentKey: key,
            // 强制更新模式下的安装，默认是IMMEDIATE 直接安装
            mandatoryInstallMode: CodePush.InstallMode.IMMEDIATE,
            //更新确认弹窗设置，设置系统自带弹窗中的内容
            updateDialog: dialog ? da : undefined,
        },
        dialog ? onStatus : undefined,
    );
}

    # 添加react-native-code-push后打包的时候无法从远程网站下载资源

    在android\build.gradle文件下添加maven { url 'https://dl.google.com/dl/android/maven2/' }

    allprojects {
    repositories {
        mavenLocal()
        maven {
            // All of React Native (JS, Obj-C sources, Android binaries) is installed from npm
            url("$rootDir/../node_modules/react-native/android")
        }
        maven {
            // Android JSC is installed from npm
            url("$rootDir/../node_modules/jsc-android/dist")
        }
        maven { url 'https://www.jitpack.io' }
        maven { url 'https://jitpack.io' }
        maven { url 'https://dl.google.com/dl/android/maven2/' }
        google()
        jcenter()
    
    }
}
